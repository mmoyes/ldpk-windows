116  virtual bool getJacobianMatrixDQ(double x0, double y0, double &m00, double &m01, double &m10, double &m11)
117  {
118  bool ok = true;
119  double eps = 1e-4,x00,x10,x01,x11,y00,y10,y01,y11;
120  ok &= undistort(x0 - eps,y0,x00,y00);
121  ok &= undistort(x0 + eps,y0,x01,y01);
122  ok &= undistort(x0,y0 - eps,x10,y10);
123  ok &= undistort(x0,y0 + eps,x11,y11);
124  m00 = (x01 - x00) / (2.0 * eps);
125  m01 = (x11 - x10) / (2.0 * eps);
126  m10 = (y01 - y00) / (2.0 * eps);
127  m11 = (y11 - y10) / (2.0 * eps);
128  return ok;
129  }
